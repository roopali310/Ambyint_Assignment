CREATE VIEW Validation (ROW_Count) AS
SELECT COUNT(*) FROM AMBYINT_DB.ASSIGNMENT.NETFLIX_DATA;


CREATE VIEW Validation_Director (SHOW_ID,TITLE ) AS
SELECT SHOW_ID, TITLE FROM AMBYINT_DB.ASSIGNMENT.NETFLIX_DATA WHERE DIRECTOR IS NULL OR DIRECTOR=' ';
CREATE VIEW Validation_Director1 (Count1 ) AS
SELECT COUNT(*) FROM AMBYINT_DB.ASSIGNMENT.NETFLIX_DATA WHERE DIRECTOR IS NULL OR DIRECTOR=' ';


CREATE VIEW Validation_CAST (SHOW_ID,TITLE ) AS
SELECT SHOW_ID, TITLE FROM AMBYINT_DB.ASSIGNMENT.NETFLIX_DATA WHERE NETFLIX_DATA.CAST IS NULL OR NETFLIX_DATA.CAST=' ';
CREATE VIEW Validation_CAST1 (COUNT1) AS
SELECT COUNT(*) FROM AMBYINT_DB.ASSIGNMENT.NETFLIX_DATA WHERE NETFLIX_DATA.CAST IS NULL OR NETFLIX_DATA.CAST=' ';


CREATE VIEW Validation_DATE_ADDED (SHOW_ID,TITLE ) AS
SELECT SHOW_ID, TITLE FROM AMBYINT_DB.ASSIGNMENT.NETFLIX_DATA WHERE NETFLIX_DATA.DATE_ADDED IS NULL OR NETFLIX_DATA.DATE_ADDED=' ';
CREATE VIEW Validation_DATE_ADDED1 (COUNT1) AS
SELECT COUNT(*) FROM AMBYINT_DB.ASSIGNMENT.NETFLIX_DATA WHERE NETFLIX_DATA.DATE_ADDED IS NULL OR NETFLIX_DATA.DATE_ADDED=' ';


CREATE VIEW Validation_DATA (SHOW_ID,TITLE ) AS
SELECT SHOW_ID, TITLE FROM AMBYINT_DB.ASSIGNMENT.NETFLIX_DATA WHERE TYPE LIKE '%[^a-Z0-9]%';
CREATE VIEW Validation_DATA1 (SHOW_ID,TITLE ) AS
SELECT SHOW_ID, TITLE FROM AMBYINT_DB.ASSIGNMENT.NETFLIX_DATA WHERE NETFLIX_DATA.CAST LIKE '%[!@#$%]%';
